``` r
expand_dataset <- function(x) {
  url <- attr(x, "url")
  if (is.null(url) || !nzchar(url)) x else loadfulldata(x)
}
```


``` r
data(NN5)
NN5 <- expand_dataset(NN5)
NN5 <- tail(NN5, 1000)
cat("Dataset: NN5\n")
```
