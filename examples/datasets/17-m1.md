``` r
expand_dataset <- function(x) {
  url <- attr(x, "url")
  if (is.null(url) || !nzchar(url)) x else loadfulldata(x)
}
```


``` r
data(m1)
m1 <- expand_dataset(m1)
cat("Dataset: m1\n")
```
